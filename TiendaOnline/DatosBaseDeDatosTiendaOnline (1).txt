	id UNIQUEIDENTIFIER PRIMARY KEY,

CREATE TABLE Cliente (
	id UNIQUEIDENTIFIER PRIMARY KEY,
	tipoIdentificacion UNIQUEIDENTIFIER NOT NULL FOREIGN KEY REFERENCES TipoIdentificacion(id),
	identificacion NVARCHAR(50) NOT NULL,
	primerNombre NVARCHAR(50) NOT NULL,
	segundoNombre NVARCHAR(50) NOT NULL,
	primerApellido NVARCHAR(50) NOT NULL,
	segundoApellido NVARCHAR(50) NOT NULL,
	correoElectronico NVARCHAR(320) NOT NULL,
	correoElectronicoConfirmado BIT NOT NULL,
	numeroTelefonoMovil NVARCHAR(50) NOT NULL,
	numeroTelefonoMovilConfirmado BIT NOT NULL,
	fechaNacimiento DATE NOT NULL
)
